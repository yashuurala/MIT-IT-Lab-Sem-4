            v2 = input;                                         //at / side input is made zero now it is 2nd opnd stored in v2
            this.textBox1.Text = string.Empty;
            this.textBox1.Text += "";

            double num1, num2;
            double.TryParse(v1, out num1);                      // string stored in the variable v1 is converted into a number (double equivalent) 
            double.TryParse(v2, out num2);

            if (op == '+')                                     //op value checked here
            {
                res = num1 + num2;                             //result stored in res
                this.textBox1.Text = res.ToString();           // res in double convert it to string
            }
            else if (op == '-')
            {
                res = num1 - num2;
                this.textBox1.Text = res.ToString();
            }
            else if (op == '*')
            {
                res = num1 * num2;
                this.textBox1.Text = res.ToString();
            }
            else if (op == '/')
            {
                res = num1 / num2;
                this.textBox1.Text = res.ToString();
            }
            else if (op == '%')
            {
                res = num1 % num2;
                this.textBox1.Text = res.ToString();
            }
            input = string.Empty;
        }
